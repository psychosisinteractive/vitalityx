418  // ------------------------------------------------------------------

420  atapi_packet[ 1] = 0x0;
421  atapi_packet[ 2] = (lba >> 24) & 0xFF;
422  atapi_packet[ 3] = (lba >> 16) & 0xFF;
423  atapi_packet[ 4] = (lba >> 8) & 0xFF;
424  atapi_packet[ 5] = (lba >> 0) & 0xFF;

432  // ------------------------------------------------------------------

434  // (III): Delay 400 nanoseconds for select to complete:
435  // ------------------------------------------------------------------
436  for(int i = 0; i < 4; i++)

441  // (V): Tell the Controller the size of buffer:
442  // ------------------------------------------------------------------
443  ide_write(channel, ATA_REG_LBA1, (words * 2) & 0xFF); // Lower Byte of Sector Size.
444  ide_write(channel, ATA_REG_LBA2, (words * 2) >> 8); // Upper Byte of Sector Size.

446  // ------------------------------------------------------------------

448

452

456  // (IX): Receiving Data:
457  // ------------------------------------------------------------------
458  for (i = 0; i < numsects; i++) {
459  ide_wait_irq(); // Wait for an IRQ.
460  if (err = ide_polling(channel, 1))
461  return err; // Polling and return if error.
462  asm("pushw %es");

472  // ------------------------------------------------------------------
473  while (ide_read(channel, ATA_REG_STATUS) & (ATA_SR_BSY | ATA_SR_DRQ))
474  ;
475 
476  return 0; // Easy, ... Isn't it?
